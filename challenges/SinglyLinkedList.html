#pragma once

template <typename T> class SinglyLl {
  public:
    SinglyLl() = default;
    SinglyLl(const SinglyLl &) = delete;
    SinglyLl(const SinglyLl &&) = delete;

    SinglyLl &operator=(const SinglyLl &) = delete;
    SinglyLl &operator=(const SinglyLl &&) = delete;

    ~SinglyLl() {
        Node *n = head;
        while (n) {
            Node *del = n;
            n = n->next;
            delete del;
            del = nullptr;
        }
        head = nullptr;
    }

    void push_back(const T arg) {
        Node *n = new Node{.data = arg, .next = nullptr};
        addNodeBack(n);
    }

    void push_front(const T arg) {
        Node *n = new Node{.data = arg, .next = nullptr};
        addNodeFront(n);
    }

    void print_list() {
        Node *n = head;
        std::cout << "\nPrinting list:\n";

        while (n) {
            std::cout << n->data;
            n = n->next;
            if (n) {
                std::cout << " -> ";
            }
        }
        std::cout << std::endl;
    }

  private:
    struct Node {
        T data;
        Node *next;
    };

    Node *head = nullptr;

    void addNodeBack(Node *const node) {
        if (!head) {
            head = node;
        } else {
            Node *n = head;
            while (n->next) {
                n = n->next;
            }
            n->next = node;
        }
    }

    void addNodeFront(Node *const node) {
        if (!head) {
            head = node;
        } else {
            node->next = head;
            head = node;
        }
    }
};
