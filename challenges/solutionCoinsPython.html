import unittest

result = 0

def solution(coins, base, i):
    global result
    n = 0
    if i < (len(coins) - 1):
        while (base + coins[i] * n) <= 200:
            base = base + coins[i] * n
            solution(coins, base, i + 1)
            base = base - coins[i] * n
            n = n + 1
    else:
        while (base + coins[i] * n) <= 200:
            base = base + coins[i] * n
            if base == 200:
                result = result + 1
            base = base - coins[i] * n
            n = n + 1

class TestCoinsFunction(unittest.TestCase):

    def setUp(self):
        self.coins = [1, 2, 5, 10, 20, 50, 100, 200]

    def test_coins(self):
        solution(self.coins, 0, 0)
        self.assertEqual(result, 73682)

unittest.main()
