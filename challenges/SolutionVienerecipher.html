#include <cctype>
#include <iostream>

namespace {
const std::string abc{"abcdefghijklmnopqrstuvwxyz"};
}

std::string cryptText(const std::string &plaintext, const std::string &key) {
  std::string retTxt;
  uint32_t counter{};

  for (char c : plaintext) {
    if (std::isalpha(c)) {
      const uint32_t rotate = key[counter % key.size()] - 'a';
      const std::string chipher = abc.substr(rotate) + abc.substr(0, rotate);
      retTxt += chipher[c - 'a'];
    } else {
      retTxt += c;
    }
    ++counter;
  }
  return retTxt;
}

int main() {
  const std::string txt{"coding is beautiful"};
  const std::string keyword{"random"};
  std::cout << cryptText(txt, keyword) << std::endl; // toqlbs if pqrugltgc
  std::cout << cryptText("attackatdawn", "lemon") << std::endl; // lxfopvefrnhr

  return 0;
}
