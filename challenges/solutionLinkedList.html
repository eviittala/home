#include <gtest/gtest.h>

#include "CircularDoubleLinkedList.h"
#include "SinglyLinkedList.h"

namespace {
TEST(LinkedList, SinglyLinkedList) {
    SinglyLl<uint64_t> sll;
    for (uint64_t i : {5, 6, 7, 8, 9, 10}) {
        sll.push_back(i);
    }
    sll.push_front(99);
    sll.push_front(98);
    sll.print_list();
}

TEST(LinkedList, CircularDoubleLinkedList) {
    CircularDLl<uint64_t> cdll;

    for (uint64_t i : {5, 6, 7, 8, 9, 10}) {
        cdll.push_back(i);
    }
    cdll.push_front(99);
    cdll.push_front(98);
    cdll.print_list();
}
} // namespace
