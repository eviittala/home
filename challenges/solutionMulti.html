#include <algorithm>
#include <ctime>
#include <iostream>
#include <vector>

void askTableAndPrintResult(const uint32_t val) {
    std::vector<uint32_t> numbers(10);
    std::generate(std::begin(numbers), std::end(numbers),
                  [i = 0]() mutable { return ++i; });
    std::srand(std::time(nullptr));

    while (not numbers.empty()) {
        const int ran = std::rand() % numbers.size();
        int answer = 0;
        std::cout << "How much is " << val << " * " << numbers.at(ran) << " = ";
        std::cin >> answer;
        if (answer == val * numbers.at(ran)) {
            std::cout << "Right" << std::endl;
            numbers.erase(std::begin(numbers) + ran);
        } else {
            std::cout << "Wrong, correct answer is: " << val * numbers.at(ran)
                      << std::endl;
            break;
        }
    }
    if (numbers.empty()) {
        std::cout << "Well done!" << std::endl;
    } else {
        std::cout << "You'll need to practice more" << std::endl;
    }
}

int main(int args, char *argv[]) {
    if (args != 2) {
        std::cout << "One number is needed" << std::endl;
    }
    askTableAndPrintResult(std::atoi(argv[1]));
    return 0;
}
