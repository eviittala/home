import random
import sys

multi = []
def makeList():
    for i in range(2, 10):
        for j in range(1, 10):
            multi.append([i, j])

def check(val, tries = 0):
    answer = int(input("{0} * {1} = ".format(val[0], val[1])))
    if answer == val[0] * val[1]:
        print("Right! ({0})".format(len(multi)))
    else:
        if tries == 0:
            print("Wrong! Please try again")
            check(val, tries + 1)
        else:
            sys.exit("You'll need to practice more!")


if __name__ == "__main__":
    makeList()
    while len(multi) > 0:
        rand = random.randrange(0, len(multi))
        check(multi.pop(rand))
    print("Congrats! You got all right!")
